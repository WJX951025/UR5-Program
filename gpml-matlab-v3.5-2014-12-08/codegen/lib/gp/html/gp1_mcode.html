1function [varargout] = gp(hyp, inf, mean, cov, lik, x, y, xs, ys)
  2% Gaussian Process inference and prediction. The gp function provides a
  3% flexible framework for Bayesian inference and prediction with Gaussian
  4% processes for scalar targets, i.e. both regression and binary
  5% classification. The prior is Gaussian process, defined through specification
  6% of its mean and covariance function. The likelihood function is also
  7% specified. Both the prior and the likelihood may have hyperparameters
  8% associated with them.
  9%
 10% Two modes are possible: training or prediction: if no test cases are
 11% supplied, then the negative log marginal likelihood and its partial
 12% derivatives w.r.t. the hyperparameters is computed; this mode is used to fit
 13% the hyperparameters. If test cases are given, then the test set predictive
 14% probabilities are returned. Usage:
 15%
 16%   training: [nlZ dnlZ          ] = gp(hyp, inf, mean, cov, lik, x, y);
 17% prediction: [ymu ys2 fmu fs2   ] = gp(hyp, inf, mean, cov, lik, x, y, xs);
 18%         or: [ymu ys2 fmu fs2 lp] = gp(hyp, inf, mean, cov, lik, x, y, xs, ys);
 19%
 20% where:
 21%
 22%   hyp      struct of column vectors of mean/cov/lik hyperparameters
 23%   inf      function specifying the inference method 
 24%   mean     prior mean function
 25%   cov      prior covariance function
 26%   lik      likelihood function
 27%   x        n by D matrix of training inputs
 28%   y        column vector of length n of training targets
 29%   xs       ns by D matrix of test inputs
 30%   ys       column vector of length nn of test targets
 31%
 32%   nlZ      returned value of the negative log marginal likelihood
 33%   dnlZ     struct of column vectors of partial derivatives of the negative
 34%               log marginal likelihood w.r.t. mean/cov/lik hyperparameters
 35%   ymu      column vector (of length ns) of predictive output means
 36%   ys2      column vector (of length ns) of predictive output variances
 37%   fmu      column vector (of length ns) of predictive latent means
 38%   fs2      column vector (of length ns) of predictive latent variances
 39%   lp       column vector (of length ns) of log predictive probabilities
 40%
 41%   post     struct representation of the (approximate) posterior
 42%            3rd output in training mode or 6th output in prediction mode
 43%            can be reused in prediction mode gp(.., cov, lik, x, post, xs,..)
 44%
 45% See also infMethods.m, meanFunctions.m, covFunctions.m, likFunctions.m.
 46%
 47% Copyright (c) by Carl Edward Rasmussen and Hannes Nickisch, 2015-03-25.
 48%                                      File automatically generated using noweb.
 49if nargin<7 || nargin>9
 50  disp('Usage: [nlZ dnlZ          ] = gp(hyp, inf, mean, cov, lik, x, y);')
 51  disp('   or: [ymu ys2 fmu fs2   ] = gp(hyp, inf, mean, cov, lik, x, y, xs);')
 52  disp('   or: [ymu ys2 fmu fs2 lp] = gp(hyp, inf, mean, cov, lik, x, y, xs, ys);')
 53  return
 54end
 55
 56if isempty(mean), mean = {@meanZero}; end                     % set default mean
 57if ischar(mean) || isa(mean, 'function_handle'), mean = {mean}; end  % make cell
 58if isempty(cov), error('Covariance function cannot be empty'); end  % no default
 59if ischar(cov) || isa(cov,'function_handle'), cov  = {cov};  end     % make cell
 60cstr = cov{1}; if isa(cstr,'function_handle'), cstr = func2str(cstr); end
 61if strcmp(cstr,'covFITC') && isfield(hyp,'xu'), cov{3} = hyp.xu; end %use hyp.xu
 62if isempty(inf)                                   % set default inference method
 63  if strcmp(cstr,'covFITC'), inf = {@infFITC}; else inf = {@infExact}; end
 64end
 65if ischar(inf), inf = str2func(inf); end          % convert into function handle
 66if ischar(inf) || isa(inf,'function_handle'), inf = {inf};  end      % make cell
 67istr = inf{1}; if isa(istr,'function_handle'), istr = func2str(istr); end
 68if strcmp(istr,'infPrior')
 69  istr = inf{2}; if isa(istr,'function_handle'), istr = func2str(istr); end
 70end
 71if strcmp(cstr,'covFITC')                           % only infFITC* are possible
 72  if isempty(strfind(istr,'infFITC')==1)
 73    error('Only infFITC* are possible inference algorithms')
 74  end
 75end                            % only one possible class of inference algorithms
 76if isempty(lik), lik = {@likGauss}; end                        % set default lik
 77if ischar(lik) || isa(lik,'function_handle'), lik = {lik};  end      % make cell
 78lstr = lik{1}; if isa(lstr,'function_handle'), lstr = func2str(lstr); end
 79
 80D = size(x,2);
 81if strcmp(cstr,'covGrid')                           % only infGrid* are possible
 82  if isempty(strfind(istr,'infGrid')==1)
 83    error('Only infGrid* are possible inference algorithms')
 84  end
 85  D = 0; xg = cov{3}; p = numel(xg); for i=1:p, D = D+size(xg{i},2); end  % dims
 86end
 87
 88if ~isfield(hyp,'mean'), hyp.mean = []; end        % check the hyp specification
 89if eval(feval(mean{:})) ~= numel(hyp.mean)
 90  error('Number of mean function hyperparameters disagree with mean function')
 91end
 92if ~isfield(hyp,'cov'), hyp.cov = []; end
 93if eval(feval(cov{:})) ~= numel(hyp.cov)
 94  error('Number of cov function hyperparameters disagree with cov function')
 95end
 96if ~isfield(hyp,'lik'), hyp.lik = []; end
 97if eval(feval(lik{:})) ~= numel(hyp.lik)
 98  error('Number of lik function hyperparameters disagree with lik function')
 99end
100
101try                                                  % call the inference method
102  % issue a warning if a classification likelihood is used in conjunction with
103  % labels different from +1 and -1
104  if strcmp(lstr,'likErf') || strcmp(lstr,'likLogistic')
105    if ~isstruct(y)
106      uy = unique(y);
107      if any( uy~=+1 & uy~=-1 )
108        warning('You try classification with labels different from {+1,-1}')
109      end
110    end
111  end
112  if nargin>7   % compute marginal likelihood and its derivatives only if needed
113    if isstruct(y)
114      post = y;            % reuse a previously computed posterior approximation
115    else
116      post = feval(inf{:}, hyp, mean, cov, lik, x, y);
117    end
118  else
119    if nargout<=1
120      [post nlZ] = feval(inf{:}, hyp, mean, cov, lik, x, y); dnlZ = {};
121    else
122      [post nlZ dnlZ] = feval(inf{:}, hyp, mean, cov, lik, x, y);
123    end
124  end
125catch
126  msgstr = lasterr;
127  if nargin>7, error('Inference method failed [%s]', msgstr); else 
128    warning('Inference method failed [%s] .. attempting to continue',msgstr)
129    dnlZ = struct('cov',0*hyp.cov, 'mean',0*hyp.mean, 'lik',0*hyp.lik);
130    varargout = {NaN, dnlZ}; return                    % continue with a warning
131  end
132end
133
134if nargin==7                                     % if no test cases are provided
135  varargout = {nlZ, dnlZ, post};    % report -log marg lik, derivatives and post
136else
137  alpha = post.alpha; L = post.L; sW = post.sW;
138  if issparse(alpha)                  % handle things for sparse representations
139    nz = alpha ~= 0;                                 % determine nonzero indices
140    if issparse(L), L = full(L(nz,nz)); end      % convert L and sW if necessary
141    if issparse(sW), sW = full(sW(nz)); end
142  else nz = true(size(alpha,1),1); end               % non-sparse representation
143  if isempty(L)                       % in case L is not provided, we compute it
144    K = feval(cov{:}, hyp.cov, x(nz,:));
145    L = chol(eye(sum(nz))+sW*sW'.*K);
146  end
147  %verify whether L contains valid Cholesky decomposition or something different
148  Lchol = isnumeric(L) && all(all(tril(L,-1)==0)&diag(L)'>0&isreal(diag(L))');
149  ns = size(xs,1);                                       % number of data points
150  if strcmp(cstr,'covGrid'), xs = covGrid('idx2dat',cov{3},xs); end  % expand xs
151  nperbatch = 1000;                       % number of data points per mini batch
152  nact = 0;                       % number of already processed test data points
153  ymu = zeros(ns,1); ys2 = ymu; fmu = ymu; fs2 = ymu; lp = ymu;   % allocate mem
154  while nact<ns               % process minibatches of test cases to save memory
155    id = (nact+1):min(nact+nperbatch,ns);               % data points to process
156    kss = feval(cov{:}, hyp.cov, xs(id,:), 'diag');              % self-variance
157    if strcmp(cstr,'covFITC')                                % cross-covariances
158      Ks = feval(cov{:}, hyp.cov, x, xs(id,:)); Ks = Ks(nz,:); % res indep. of x
159    else
160      Ks = feval(cov{:}, hyp.cov, x(nz,:), xs(id,:));        % avoid computation
161    end
162    ms = feval(mean{:}, hyp.mean, xs(id,:));
163    N = size(alpha,2);  % number of alphas (usually 1; more in case of sampling)
164    Fmu = repmat(ms,1,N) + Ks'*full(alpha(nz,:));        % conditional mean fs|f
165    fmu(id) = sum(Fmu,2)/N;                                   % predictive means
166    if Lchol    % L contains chol decomp => use Cholesky parameters (alpha,sW,L)
167      V  = L'\(repmat(sW,1,length(id)).*Ks);
168      fs2(id) = kss - sum(V.*V,1)';                       % predictive variances
169    else                % L is not triangular => use alternative parametrisation
170      if isnumeric(L), LKs = L*Ks; else LKs = L(Ks); end    % matrix or callback
171      fs2(id) = kss + sum(Ks.*LKs,1)';                    % predictive variances
172    end
173    fs2(id) = max(fs2(id),0);   % remove numerical noise i.e. negative variances
174    Fs2 = repmat(fs2(id),1,N);     % we have multiple values in case of sampling
175    if nargin<9
176      [Lp, Ymu, Ys2] = feval(lik{:},hyp.lik,[],   Fmu(:),Fs2(:));
177    else
178      Ys = repmat(ys(id),1,N);
179      [Lp, Ymu, Ys2] = feval(lik{:},hyp.lik,Ys(:),Fmu(:),Fs2(:));
180    end
181    lp(id)  = sum(reshape(Lp, [],N),2)/N;    % log probability; sample averaging
182    ymu(id) = sum(reshape(Ymu,[],N),2)/N;          % predictive mean ys|y and ..
183    ys2(id) = sum(reshape(Ys2,[],N),2)/N;                          % .. variance
184    nact = id(end);          % set counter to index of last processed data point
185  end
186  if nargin<9
187    varargout = {ymu, ys2, fmu, fs2, [], post};        % assign output arguments
188  else
189    varargout = {ymu, ys2, fmu, fs2, lp, post};
190  end
191end
